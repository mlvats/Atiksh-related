    <script>
        const apiKey = "";
        let currentBlueprintData = null;

        // Initialize Lucide Icons
        lucide.createIcons();

        // Utility: Exponential Backoff Fetch
        async function geminiFetch(prompt, systemPrompt, isJson = false) {
            const delays = [1000, 2000, 4000, 8000];
            const payload = {
                contents: [{ parts: [{ text: prompt }] }],
                systemInstruction: { parts: [{ text: systemPrompt }] },
                tools: [{ "google_search": {} }]
            };

            for (let i = 0; i <= delays.length; i++) {
                try {
                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    if (!response.ok) throw new Error('API Fail');
                    const data = await response.json();
                    const text = data.candidates[0].content.parts[0].text;
                    
                    if (isJson) {
                        const jsonStr = text.substring(text.indexOf('{'), text.lastIndexOf('}') + 1);
                        return JSON.parse(jsonStr);
                    }
                    return text;
                } catch (e) {
                    if (i === delays.length) throw e;
                    await new Promise(r => setTimeout(r, delays[i]));
                }
            }
        }

        // Feature: Provider Locator
        document.getElementById('locator-form').addEventListener('submit', async (e) => {
            e.preventDefault();
            const diag = document.getElementById('diagnosis-input').value;
            const zip = document.getElementById('zip-input').value;
            if (!diag || !zip) return;

            const loader = document.getElementById('locator-loader');
            const btnText = document.getElementById('locator-btn-text');
            const resultsDiv = document.getElementById('locator-results');

            loader.classList.remove('hidden');
            btnText.classList.add('hidden');
            resultsDiv.innerHTML = '';

            try {
                const system = `Return JSON only: { categories: [{ title, description, providers: [{ name, address }] }] }`;
                const prompt = `Find real clinical resources for ${diag} in zip code ${zip}.`;
                const data = await geminiFetch(prompt, system, true);

                resultsDiv.innerHTML = data.categories.map(cat => `
                    <div class="bg-white rounded-[2rem] p-8 text-slate-900 border-t-8 border-indigo-500 shadow-xl">
                        <h4 class="font-black text-xl mb-2">${cat.title}</h4>
                        <p class="text-sm text-slate-500 mb-6">${cat.description}</p>
                        <div class="space-y-3">
                            ${cat.providers.map(p => `
                                <div class="p-4 bg-slate-50 rounded-xl border border-slate-100">
                                    <p class="font-bold text-sm">${p.name}</p>
                                    <p class="text-xs text-slate-400 mt-1">${p.address}</p>
                                </div>
                            `).join('')}
                        </div>
                    </div>
                `).join('');
            } catch (err) {
                resultsDiv.innerHTML = `<p class="text-white text-center col-span-2">Search failed. Please try again.</p>`;
            } finally {
                loader.classList.add('hidden');
                btnText.classList.remove('hidden');
            }
        });

        // Feature: Blueprint Generator
        document.getElementById('blueprint-form').addEventListener('submit', async (e) => {
            e.preventDefault();
            const income = document.getElementById('income-input').value;
            if (!income) return;

            addChatMessage('user', `Generate a blueprint for $${income} annual income.`);
            
            try {
                const system = `Expert planner. Output JSON: { income, headline, summary, allocations: [{category, amount, percentage, rationale}], legalSteps: [] }`;
                const prompt = `Analyze $${income} annual household income. Create special needs financial roadmap.`;
                const data = await geminiFetch(prompt, system, true);
                currentBlueprintData = data;

                renderBlueprint(data);
                addChatMessage('ai', "Blueprint generated! You can view and download it below.");
                document.getElementById('blueprint-viewer').classList.remove('hidden');
                document.getElementById('blueprint-viewer').scrollIntoView();
            } catch (err) {
                addChatMessage('ai', "Failed to generate blueprint. Please try again.");
            }
        });

        function addChatMessage(role, text) {
            const container = document.getElementById('chat-messages');
            const div = document.createElement('div');
            div.className = `flex ${role === 'user' ? 'justify-end' : 'justify-start'}`;
            div.innerHTML = `
                <div class="max-w-[85%] p-6 rounded-3xl shadow-sm ${role === 'user' ? 'bg-slate-900 text-white chat-bubble-user' : 'bg-white text-slate-900 border chat-bubble-ai'}">
                    ${text}
                </div>
            `;
            container.appendChild(div);
            container.scrollTop = container.scrollHeight;
        }

        function renderBlueprint(data) {
            document.getElementById('bp-headline').textContent = data.headline;
            
            const grid = document.getElementById('bp-allocations-grid');
            grid.innerHTML = data.allocations.map(a => `
                <div class="bg-slate-50 border p-8 rounded-[2rem]">
                    <h4 class="text-xs font-black text-slate-400 uppercase tracking-widest mb-4">${a.category}</h4>
                    <p class="text-3xl font-black mb-1">${a.amount}</p>
                    <div class="h-1.5 w-full bg-slate-200 rounded-full mb-4">
                        <div class="h-full bg-indigo-600 rounded-full" style="width: ${a.percentage}"></div>
                    </div>
                    <p class="text-xs text-slate-500">${a.rationale}</p>
                </div>
            `).join('');

            const legal = document.getElementById('bp-legal-list');
            legal.innerHTML = data.legalSteps.map((s, i) => `
                <div class="flex gap-4 items-start bg-white/5 p-5 rounded-2xl border border-white/10">
                    <div class="w-8 h-8 rounded-full bg-indigo-500 flex items-center justify-center shrink-0 font-bold">${i+1}</div>
                    <p class="text-sm opacity-90">${s}</p>
                </div>
            `).join('');

            // Set Print Content
            document.getElementById('print-content').innerHTML = `
                <h2 class="text-2xl font-bold mb-4">${data.headline}</h2>
                <p class="mb-8">Income Basis: ${data.income}</p>
                <div class="grid grid-cols-2 gap-6 mb-10">
                    ${data.allocations.map(a => `
                        <div class="border p-4 rounded-xl">
                            <h4 class="font-bold text-indigo-700">${a.category}</h4>
                            <p class="text-xl font-black">${a.amount} (${a.percentage})</p>
                            <p class="text-[10px] text-slate-500">${a.rationale}</p>
                        </div>
                    `).join('')}
                </div>
                <div class="bg-slate-50 p-6 rounded-xl">
                    <h4 class="font-bold mb-4">Required Legal Actions</h4>
                    ${data.legalSteps.map(s => `<p class="text-sm mb-2">• ${s}</p>`).join('')}
                </div>
            `;
        }

        function downloadPDF() {
            window.print();
        }

import React, { useState, useEffect, useRef } from 'react';
import { 
  Heart, 
  MapPin, 
  ShieldCheck, 
  LineChart, 
  Search, 
  MessageSquare, 
  Users, 
  ChevronRight,
  BrainCircuit,
  Wallet,
  Building2,
  Stethoscope,
  Baby,
  GraduationCap,
  Puzzle,
  HandHeart,
  CalendarDays,
  Sparkles,
  Info,
  ExternalLink,
  Activity,
  School,
  Library,
  Loader2,
  AlertCircle,
  FileText,
  DollarSign,
  ArrowRight,
  Phone,
  Download,
  CheckCircle2,
  PieChart,
  Target
} from 'lucide-react';

const apiKey = ""; // Provided at runtime

const App = () => {
  const [zipCode, setZipCode] = useState('');
  const [diagnosis, setDiagnosis] = useState(''); 
  const [isSearching, setIsSearching] = useState(false);
  const [results, setResults] = useState(null);
  const [chatInput, setChatInput] = useState('');
  const [annualIncomeInput, setAnnualIncomeInput] = useState('');
  const [isGeneratingBlueprint, setIsGeneratingBlueprint] = useState(false);
  const [currentBlueprint, setCurrentBlueprint] = useState(null);
  const [chatHistory, setChatHistory] = useState([
    { role: 'assistant', text: 'Hello. I am your SpecialtyAI financial guide. How can I help you plan for your child\'s future today? We can discuss ABLE accounts, Special Needs Trusts, or government benefit coordination.' }
  ]);
  const [isTyping, setIsTyping] = useState(false);
  const chatEndRef = useRef(null);
  const blueprintRef = useRef(null);

  const scrollToBottom = () => {
    chatEndRef.current?.scrollIntoView({ behavior: "smooth" });
  };

  useEffect(() => {
    scrollToBottom();
  }, [chatHistory, isTyping]);

  const callGemini = async (prompt, systemPrompt, isJson = false) => {
    let retries = 0;
    const delays = [1000, 2000, 4000, 8000, 16000];

    const payload = {
      contents: [{ parts: [{ text: prompt }] }],
      systemInstruction: { parts: [{ text: systemPrompt }] },
      tools: [{ "google_search": {} }]
    };

    while (retries < 5) {
      try {
        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload)
        });

        if (!response.ok) throw new Error(`API Error: ${response.status}`);
        
        const data = await response.json();
        let text = data.candidates?.[0]?.content?.parts?.[0]?.text;
        if (!text) throw new Error("Empty response");

        let parsedContent = text;
        if (isJson) {
          try {
            const firstBrace = text.indexOf('{');
            const lastBrace = text.lastIndexOf('}');
            const jsonString = text.substring(firstBrace, lastBrace + 1);
            parsedContent = JSON.parse(jsonString);
          } catch (e) {
            throw new Error("Format error");
          }
        }

        return {
          content: parsedContent,
          sources: data.candidates?.[0]?.groundingMetadata?.groundingAttributions?.map(a => ({ uri: a.web?.uri, title: a.web?.title })) || []
        };
      } catch (error) {
        retries++;
        if (retries === 5) throw error;
        await new Promise(res => setTimeout(res, delays[retries - 1]));
      }
    }
  };

  const handleGenerateBlueprint = async (e) => {
    e.preventDefault();
    if (!annualIncomeInput || isNaN(annualIncomeInput)) return;
    
    setIsGeneratingBlueprint(true);
    const income = annualIncomeInput;
    const userMsg = `I have an annual household income of $${income}. Generate a detailed financial blueprint. Output MUST be in JSON format.
    JSON Structure: {
      "income": "$${income}",
      "headline": "Short title for the plan",
      "summary": "2 sentence executive summary",
      "allocations": [
        {"category": "ABLE Account", "amount": "$X/mo", "percentage": "X%", "rationale": "Why this amount"},
        {"category": "Special Needs Trust", "amount": "$X/mo", "percentage": "X%", "rationale": "Why this amount"},
        {"category": "Emergency Fund", "amount": "$X/mo", "percentage": "X%", "rationale": "Why this amount"},
        {"category": "Care & Living", "amount": "$X/mo", "percentage": "X%", "rationale": "Why this amount"}
      ],
      "legalSteps": ["Step 1", "Step 2", "Step 3"]
    }`;
    
    setChatHistory(prev => [...prev, { role: 'user', text: `Generate a visual blueprint for $${income} annual income.` }]);
    
    try {
      const systemPrompt = `You are a professional special needs financial planner. You analyze income and provide structured JSON data for a visual blueprint. Ensure the amounts are mathematically sound for the given annual income.`;
      const response = await callGemini(userMsg, systemPrompt, true);
      setCurrentBlueprint(response.content);
      setChatHistory(prev => [...prev, { role: 'assistant', text: "I have successfully generated your custom Financial Blueprint. You can view the full visual roadmap and download the PDF in the 'Active Blueprint' section below." }]);
      setAnnualIncomeInput('');
      
      // Scroll to the new blueprint section after a short delay
      setTimeout(() => {
        document.getElementById('blueprint-viewer')?.scrollIntoView({ behavior: 'smooth' });
      }, 500);
    } catch (err) {
      setChatHistory(prev => [...prev, { role: 'assistant', text: "I had trouble generating that specific data structure. Please try again." }]);
    } finally {
      setIsGeneratingBlueprint(false);
    }
  };

  const downloadBlueprintPDF = () => {
    const originalTitle = document.title;
    document.title = `SpecialtyAI_Blueprint_${annualIncomeInput || 'Export'}`;
    window.print();
    document.title = originalTitle;
  };

  const handleServiceSearch = async (e) => {
    e.preventDefault();
    if (!zipCode || zipCode.length < 5 || !diagnosis) return;
    setIsSearching(true);
    setResults(null); 
    
    const systemPrompt = `You are a specialized local service locator for families with disabilities. 
    Analyze the request and return ONLY a valid JSON object. 
    You must provide REAL, EXISTING providers with their FULL ADDRESSES and phone numbers if available.
    JSON Structure: { 
      "summary": "A 2-3 sentence overview of what is available in the specific zip code.", 
      "categories": [{
        "title": "Service Type (e.g. ABA Therapy, Speech Pathology)", 
        "description": "Specific details about availability and focus.", 
        "providers": [
          { "name": "Provider Name", "address": "Full Street Address, City, State, Zip", "phone": "Phone Number" }
        ]
      }] 
    }`;
    const userPrompt = `Find local clinical and support resources for ${diagnosis} near zip code ${zipCode}. I need specific names of businesses/clinics and their exact addresses for navigation.`;

    try {
      const response = await callGemini(userPrompt, systemPrompt, true);
      setResults(response);
    } catch (err) {
      setResults({ error: "The search encountered an error. Please check your connection and try again." });
    } finally {
      setIsSearching(false);
    }
  };

  const handleChat = async (e) => {
    e.preventDefault();
    if (!chatInput.trim()) return;
    const userMsg = chatInput;
    setChatHistory(prev => [...prev, { role: 'user', text: userMsg }]);
    setChatInput('');
    setIsTyping(true);
    try {
      const response = await callGemini(userMsg, "Expert special needs financial guide.");
      setChatHistory(prev => [...prev, { role: 'assistant', text: response.content }]);
    } catch (err) {
      setChatHistory(prev => [...prev, { role: 'assistant', text: "Error connecting. Try again." }]);
    } finally {
      setIsTyping(false);
    }
  };

  const getCategoryIcon = (type) => {
    const t = (type || "").toLowerCase();
    if (t.includes('therapy') || t.includes('speech')) return <Stethoscope className="text-blue-500" />;
    if (t.includes('school')) return <School className="text-purple-500" />;
    if (t.includes('puzzle') || t.includes('social')) return <Puzzle className="text-orange-500" />;
    return <Building2 className="text-slate-500" />;
  };

  return (
    <div className="min-h-screen bg-slate-50 text-slate-900 font-sans selection:bg-indigo-100">
      {/* Print Overlay - Hidden in Browser */}
      <div className="hidden print:block p-12 bg-white min-h-screen">
        <div className="flex items-center gap-3 mb-10 border-b-2 border-indigo-600 pb-6">
           <BrainCircuit className="text-indigo-600 w-12 h-12" />
           <div>
             <h1 className="text-3xl font-black text-slate-900">SpecialtyAI</h1>
             <p className="text-sm font-bold text-indigo-600 uppercase tracking-widest">Financial Roadmap & Care Blueprint</p>
           </div>
        </div>
        
        {currentBlueprint && (
          <div className="space-y-8">
            <div>
              <h2 className="text-2xl font-bold mb-2">{currentBlueprint.headline}</h2>
              <p className="text-slate-600">Generated for Annual Household Income: <span className="font-bold text-slate-900">{currentBlueprint.income}</span></p>
            </div>
            
            <div className="grid grid-cols-2 gap-4">
              {currentBlueprint.allocations.map((a, i) => (
                <div key={i} className="border p-4 rounded-xl">
                  <h4 className="font-bold text-indigo-700 mb-1">{a.category}</h4>
                  <p className="text-xl font-black">{a.amount} <span className="text-sm font-normal text-slate-500">({a.percentage})</span></p>
                  <p className="text-xs text-slate-500 mt-2">{a.rationale}</p>
                </div>
              ))}
            </div>

            <div className="bg-slate-50 p-6 rounded-xl border border-slate-200">
              <h4 className="font-bold mb-4">Required Legal & Administrative Actions</h4>
              <ul className="space-y-2">
                {currentBlueprint.legalSteps.map((s, i) => (
                  <li key={i} className="text-sm flex gap-2">
                    <span className="font-bold text-indigo-600">{i + 1}.</span> {s}
                  </li>
                ))}
              </ul>
            </div>
            
            <div className="mt-20 pt-10 border-t text-[10px] text-slate-400 italic">
              Disclaimer: This document is for informational purposes only. Consult with a qualified Special Needs Attorney or CFP before making financial decisions. Generated by SpecialtyAI Professional Platform.
            </div>
          </div>
        )}
      </div>

      <nav className="sticky top-0 z-50 bg-white/80 backdrop-blur-md border-b border-slate-200 print:hidden">
        <div className="max-w-7xl mx-auto px-4 h-16 flex items-center justify-between">
          <div className="flex items-center gap-2">
            <div className="bg-indigo-600 p-2 rounded-lg"><BrainCircuit className="text-white w-6 h-6" /></div>
            <span className="text-xl font-bold bg-clip-text text-transparent bg-gradient-to-r from-indigo-600 to-blue-500">SpecialtyAI</span>
          </div>
          <div className="hidden md:flex gap-8 text-sm font-medium text-slate-600">
            <a href="#locator" className="hover:text-indigo-600 transition-colors">Resources</a>
            <a href="#planner" className="hover:text-indigo-600 transition-colors">AI Planner</a>
          </div>
          <button className="bg-indigo-600 text-white px-5 py-2 rounded-full text-sm font-semibold hover:bg-indigo-700 shadow-md">Get Started</button>
        </div>
      </nav>

      {/* Hero Section */}
      <section className="relative bg-white pt-20 pb-24 px-4 overflow-hidden border-b border-slate-100 print:hidden">
        <div className="max-w-7xl mx-auto grid lg:grid-cols-2 gap-12 items-center">
          <div className="text-left">
            <div className="inline-flex items-center gap-2 px-4 py-2 rounded-full bg-indigo-50 text-indigo-700 text-sm font-medium mb-8">
              <Sparkles size={16} /> Comprehensive Financial & Clinical Care
            </div>
            <h1 className="text-5xl md:text-6xl font-extrabold tracking-tight text-slate-900 mb-6 leading-tight">
              Secure their future with <br />
              <span className="text-indigo-600 italic">Intelligent Planning</span>
            </h1>
            <p className="text-xl text-slate-600 mb-10 leading-relaxed">
              SpecialtyAI combines empathetic financial expertise with advanced AI to guide families through complex milestones and care transitions.
            </p>
            <div className="flex flex-wrap gap-4">
               <button className="bg-indigo-600 text-white px-8 py-4 rounded-2xl font-bold hover:bg-indigo-700 transition-all shadow-lg shadow-indigo-200">Start Planning</button>
               <button className="bg-white border-2 border-slate-100 text-slate-700 px-8 py-4 rounded-2xl font-bold hover:bg-slate-50 transition-all">Watch Demo</button>
            </div>
          </div>
          <div className="relative">
            <img 
              src="https://images.unsplash.com/photo-1594608661623-aa0bd3a69d98?auto=format&fit=crop&q=80&w=800" 
              alt="Parent and child interacting warmly" 
              className="rounded-[3rem] shadow-2xl z-10 relative object-cover h-[500px] w-full border-8 border-white"
            />
          </div>
        </div>
      </section>

      {/* Active Blueprint Viewer Section - New Section */}
      {currentBlueprint && (
        <section id="blueprint-viewer" className="py-24 bg-white border-b border-slate-100 print:hidden">
           <div className="max-w-7xl mx-auto px-4">
             <div className="flex flex-col md:flex-row md:items-end justify-between gap-6 mb-12">
               <div>
                  <div className="inline-flex items-center gap-2 px-3 py-1 rounded-full bg-green-50 text-green-700 text-[10px] font-black uppercase tracking-widest mb-4 border border-green-200">
                    <CheckCircle2 size={12}/> Analysis Complete
                  </div>
                  <h2 className="text-4xl font-black text-slate-900 tracking-tight">{currentBlueprint.headline}</h2>
                  <p className="text-slate-500 font-medium mt-2">Custom allocations based on <span className="text-indigo-600 font-bold">{currentBlueprint.income}</span> annual income.</p>
               </div>
               <button 
                onClick={downloadBlueprintPDF}
                className="bg-slate-900 text-white px-8 py-4 rounded-2xl font-black text-sm uppercase tracking-widest hover:bg-black transition-all flex items-center justify-center gap-3 shadow-xl active:scale-95 shrink-0"
               >
                 <Download size={20} /> Download PDF Blueprint
               </button>
             </div>

             <div className="grid grid-cols-1 lg:grid-cols-4 gap-6 mb-12">
               {currentBlueprint.allocations.map((item, idx) => (
                 <div key={idx} className="bg-slate-50 border border-slate-200 p-8 rounded-[2rem] hover:bg-white hover:border-indigo-400 hover:shadow-xl transition-all group">
                    <div className="bg-white p-3 rounded-xl shadow-sm w-fit mb-6 group-hover:bg-indigo-50 transition-colors">
                      <Target size={24} className="text-indigo-600" />
                    </div>
                    <h4 className="text-sm font-black text-slate-400 uppercase tracking-widest mb-1">{item.category}</h4>
                    <div className="flex items-baseline gap-2 mb-4">
                      <span className="text-3xl font-black text-slate-900">{item.amount}</span>
                      <span className="text-sm font-bold text-indigo-600">/mo</span>
                    </div>
                    <div className="h-1.5 w-full bg-slate-200 rounded-full mb-6 overflow-hidden">
                      <div className="h-full bg-indigo-600 rounded-full" style={{ width: item.percentage }}></div>
                    </div>
                    <p className="text-sm text-slate-600 leading-relaxed">{item.rationale}</p>
                 </div>
               ))}
             </div>

             <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
               <div className="lg:col-span-2 bg-indigo-900 rounded-[2.5rem] p-10 text-white shadow-2xl relative overflow-hidden">
                  <div className="relative z-10">
                    <h3 className="text-2xl font-bold mb-8 flex items-center gap-3">
                      <ShieldCheck className="text-indigo-300" /> Legal & Compliance Roadmap
                    </h3>
                    <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                      {currentBlueprint.legalSteps.map((step, i) => (
                        <div key={i} className="flex gap-4 items-start bg-white/5 p-5 rounded-2xl border border-white/10">
                           <div className="w-8 h-8 rounded-full bg-indigo-500 flex items-center justify-center shrink-0 font-black text-xs">{i+1}</div>
                           <p className="text-sm font-medium leading-relaxed opacity-90">{step}</p>
                        </div>
                      ))}
                    </div>
                  </div>
                  <div className="absolute top-0 right-0 w-64 h-64 bg-indigo-400/20 rounded-full blur-3xl -mr-32 -mt-32"></div>
               </div>

               <div className="bg-white border-2 border-slate-100 rounded-[2.5rem] p-10 flex flex-col items-center text-center justify-center">
                 <div className="bg-indigo-50 p-6 rounded-full mb-6">
                    <PieChart size={48} className="text-indigo-600" />
                 </div>
                 <h4 className="text-xl font-bold mb-2">Benefit Eligibility</h4>
                 <p className="text-sm text-slate-500 mb-6">This allocation is optimized to maintain SSI and Medicaid thresholds.</p>
                 <div className="w-full flex justify-between px-6 py-4 bg-slate-50 rounded-2xl mb-2">
                    <span className="text-xs font-black uppercase text-slate-400">Status</span>
                    <span className="text-xs font-black uppercase text-green-600">Compliant</span>
                 </div>
                 <div className="w-full flex justify-between px-6 py-4 bg-slate-50 rounded-2xl">
                    <span className="text-xs font-black uppercase text-slate-400">Risk Profile</span>
                    <span className="text-xs font-black uppercase text-indigo-600">Conservative</span>
                 </div>
               </div>
             </div>
           </div>
        </section>
      )}

      {/* Support & Locator Section */}
      <section id="locator" className="py-20 px-4 bg-slate-50 print:hidden">
        <div className="max-w-7xl mx-auto space-y-8">
          {/* Clinical Locator Card */}
          <div className="bg-indigo-900 rounded-[3rem] p-8 lg:p-14 text-white shadow-2xl relative overflow-hidden flex flex-col min-h-[600px]">
            <div className="relative z-10 w-full">
              <div className="max-w-3xl mb-12">
                <div className="inline-flex items-center gap-2 px-3 py-1 rounded-full bg-indigo-500/30 text-indigo-200 text-xs font-bold tracking-widest uppercase mb-6 border border-indigo-400/20">
                  <MapPin size={12}/> Regional Care Network
                </div>
                <h2 className="text-4xl font-extrabold mb-5 text-white tracking-tight">Clinical & Support Locator</h2>
                <p className="text-xl text-indigo-100 leading-relaxed opacity-90">Find vetted providers in your area. We scan for specific clinical specialties, behavioral centers, and respite care facilities with exact location data.</p>
              </div>
              
              <form onSubmit={handleServiceSearch} className="mb-14 max-w-5xl">
                <div className="bg-white/10 backdrop-blur-xl rounded-[2rem] p-3 flex flex-col lg:row gap-3 border border-white/10 shadow-2xl">
                  <div className="flex-1 flex items-center px-6 gap-4 min-h-[64px]">
                    <Activity className="text-indigo-300 shrink-0" size={24} />
                    <input 
                      type="text" 
                      placeholder="Diagnosis or Service (e.g. ABA Therapy, Sensory Gym)" 
                      className="w-full bg-transparent border-none focus:ring-0 text-white placeholder-indigo-200/60 font-medium text-lg"
                      value={diagnosis}
                      onChange={(e) => setDiagnosis(e.target.value)}
                    />
                  </div>
                  <div className="w-full lg:w-56 flex items-center px-6 gap-4 lg:border-l border-white/20 min-h-[64px]">
                    <MapPin className="text-indigo-300 shrink-0" size={24} />
                    <input 
                      type="text" 
                      placeholder="Zip Code" 
                      className="w-full bg-transparent border-none focus:ring-0 text-white placeholder-indigo-200/60 font-bold text-lg"
                      value={zipCode}
                      onChange={(e) => setZipCode(e.target.value)}
                      maxLength={5}
                    />
                  </div>
                  <button 
                    type="submit" 
                    disabled={isSearching}
                    className="bg-white text-indigo-900 px-10 py-5 rounded-2xl font-black text-sm uppercase tracking-widest hover:bg-indigo-50 transition-all flex items-center justify-center gap-3 shadow-xl shrink-0 active:scale-95"
                  >
                    {isSearching ? <Loader2 className="animate-spin" size={20} /> : <><Search size={20} /> Find Providers</>}
                  </button>
                </div>
              </form>

              <div className="w-full overflow-y-auto pr-2 custom-scrollbar min-h-[400px]">
                {isSearching && (
                  <div className="flex flex-col items-center justify-center py-32 text-indigo-100">
                    <Loader2 className="animate-spin mb-8" size={64} />
                    <p className="text-2xl font-bold tracking-tight mb-2">Locating Verified Facilities</p>
                  </div>
                )}

                {results?.content && (
                  <div className="space-y-10 animate-in fade-in slide-in-from-bottom-8 duration-700 pb-20">
                    <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
                      {results.content.categories?.map((cat, idx) => (
                        <div key={idx} className="bg-white rounded-[2.5rem] p-8 lg:p-10 shadow-2xl text-slate-900 border-t-[12px] border-indigo-500">
                          <div className="flex items-center gap-5 mb-6">
                            <div className="p-4 bg-indigo-50 rounded-2xl text-indigo-600">{getCategoryIcon(cat.title)}</div>
                            <h4 className="font-black text-2xl text-slate-900 tracking-tight leading-none">{cat.title}</h4>
                          </div>
                          <p className="text-slate-600 mb-8 leading-relaxed font-medium">{cat.description}</p>
                          <div className="space-y-4 pt-8 border-t border-slate-100">
                             {cat.providers?.map((provider, pIdx) => (
                               <div key={pIdx} className="bg-slate-50 border border-slate-100 rounded-2xl p-5 hover:border-indigo-200 transition-colors group">
                                 <h5 className="font-bold text-slate-800">{provider.name}</h5>
                                 <div className="flex items-start gap-2 text-sm text-slate-500 mt-2">
                                   <MapPin size={14} className="mt-1 shrink-0 text-indigo-400" />
                                   <span className="leading-tight">{provider.address}</span>
                                 </div>
                               </div>
                             ))}
                          </div>
                        </div>
                      ))}
                    </div>
                  </div>
                )}
              </div>
            </div>
          </div>
        </div>
      </section>

      {/* AI Planner Section */}
      <section id="planner" className="py-24 bg-white relative print:hidden">
        <div className="max-w-7xl mx-auto px-4">
          <div className="grid lg:grid-cols-2 gap-16 items-start">
            
            {/* Chat Interface */}
            <div className="bg-white border border-slate-200 rounded-[3rem] shadow-2xl flex flex-col h-[780px] overflow-hidden border-indigo-100 ring-1 ring-slate-200/50">
              <div className="p-6 border-b bg-indigo-600 text-white flex justify-between items-center">
                <div className="flex flex-col">
                  <span className="font-bold">SpecialtyAI Planning Assistant</span>
                  <span className="text-[10px] opacity-90 uppercase font-black tracking-widest mt-1">Financial Guardrails Active</span>
                </div>
                <Info size={20} className="opacity-80" />
              </div>
              
              <div className="flex-1 overflow-y-auto p-8 space-y-6 bg-slate-50/40 custom-scrollbar-light">
                {chatHistory.map((msg, idx) => (
                  <div key={idx} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
                    <div className={`max-w-[85%] p-6 rounded-3xl ${msg.role === 'user' ? 'bg-slate-900 text-white rounded-tr-none shadow-xl' : 'bg-white text-slate-900 border border-indigo-100 shadow-md'}`}>
                      {msg.text.split('\n').map((line, i) => (
                        <p key={i} className={`text-[15px] leading-relaxed mb-3 last:mb-0`}>{line}</p>
                      ))}
                    </div>
                  </div>
                ))}
                {isTyping && (
                  <div className="flex gap-3 items-center text-indigo-600 font-black text-[10px] ml-4 animate-pulse uppercase tracking-widest">
                    <Loader2 size={14} className="animate-spin" /> Analyzing Financial Matrix...
                  </div>
                )}
                <div ref={chatEndRef} />
              </div>

              {/* Blueprint Action Bar */}
              <div className="px-8 py-8 bg-indigo-50/50 border-t border-indigo-100">
                <div className="flex flex-col gap-5">
                  <div className="flex items-center gap-3">
                    <div className="p-2 bg-indigo-100 rounded-xl"><FileText size={20} className="text-indigo-700" /></div>
                    <span className="text-sm font-black text-indigo-900 uppercase tracking-tighter">Blueprint Roadmap Generator</span>
                  </div>
                  
                  <form onSubmit={handleGenerateBlueprint} className="flex flex-col sm:flex-row gap-3">
                    <div className="relative flex-1 group">
